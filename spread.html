Spread and Rest (...)

spread.ts

							
	function add(x,y,z){
		return x + y + z;
	}

	var result = add(...[1,2,3]);

spread.js

							
	function add(x, y, z) {
		return x + y + z;
	}
	var result = add.apply(void 0, [1, 2, 3]);
When the array is passed as a function to a parameter, it spreads out into invidual values.

rest.ts

							
	function addAll(x, y,...numbers){
		var sum = x + y;
		numbers.forEach(function(arg){
			sum += arg;
		});
		return sum;
	}
	var result1 = addAll(1,2,3,4,5);

Parameters 1 and 2 get assigned to x and y

The ...numbers parameter gets the rest of the arguments into an array.

rest.js

							
	function addAll(x, y) {
		var numbers = [];
		for (var _i = 2; _i < arguments.length; _i++) {
			numbers[_i - 2] = arguments[_i];
		}
		var sum = x + y;
		numbers.forEach(function (arg) {
			sum += arg;
		});
		return sum;
	}
	var result1 = addAll(1,2,3,4,5);
