Using var now

In ES5, variables are

  1. function scoped.
  2. hoisted to the to the top of the function.

var.ts

							
	var x = 3;
	function getRandomNumber(randomize:boolean) {
		if (randomize) {
			var x = Math.random();
			return x;
		}
		return x;
	}
	var result = getRandomNumber(false);

Now Refactoring using let

let.ts


	let x = 3;
	function getRandomNumber(randomize:boolean) {
		if (randomize) {
			let x = Math.random();
			return x;
		}
		return x;
	}
	var result = getRandomNumber(false);
