• Modules are excuted in their own scope
  • All variables and functions are not visible outsode the file, Unless
  • It is decorated with export - Visible outside of this file
  • The consuming libray imports the var/functions.
  • You need Module Loader- To Load the modules as needed.
  • SystemJS, Webpack etc.
  • Modules are loaded Async

Modules- lib.ts

							

		//private function
		function youCantCallMe(){
           toast.info("Inside youCantCallMe ");
		}
		//public function
    	export const sqrt=Math.sqrt;
    	export function square(x){
        	return x * x;
    	}

Modules - lib.js

							

	"use strict";
	//private function
	function youCantCallMe() {
		toastr.info("Inside youCantCallMe ");
	}
	exports.sqrt = Math.sqrt;
	function square(x) {
		return x * x;
	}
	exports.square = square;

Modules- main.ts

							

    import {sqrt,square} from './lib';
    toastr.info(square(5));

Modules - main.js

							
	"use strict";
    var lib_1 = require('./lib');
    toastr.info(lib_1.square(5));
